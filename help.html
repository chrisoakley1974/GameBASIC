DIM A(10)
FOR I = 0 TO 10 STEP 2
  A(I) = I
NEXT I
IF X > 0 THEN
  PRINT("HI")
ELSEIF X = 0 THEN
  PRINT("ZERO")
ELSE
  PRINT("NEG")
ENDIF
WHILE X < 10
  LET X = X + 1
ENDWHILE
END
